532 private List<WinGetSource> ExportStringToSources(string exportString)
533 {
534 List<WinGetSource> sourceList = new List<WinGetSource>();
535
536 string[] jsonStrings = exportString.Split("}{");
537 StringBuilder jsonString = new StringBuilder();
538 for (int i = 0; i < jsonStrings.Length; i++)
539 {
540 if (!jsonStrings[i].StartsWith('{'))
541 {
542 jsonString.Append('{');
543 }
544
545 jsonString.Append(jsonStrings[i]);
546
547 if (!jsonStrings[i].EndsWith('}'))
548 {
549 jsonString.Append('}');
550 }
551
552 WinGetSource? source =
553 JsonHandler.StringToObject<WinGetSource>(jsonString.ToString());
554
555 if (source == null)
556 {
557 throw new WinGetActionFailedException("Exporting sources failed. Could not parse json.");
558 }
559
560 sourceList.Add(source);
561
562 jsonString.Clear();
563 }
564
565 return sourceList;
566 }

583 return false;
