558 private List<WinGetSource> ExportStringToSources(string exportString)
559 {
560 List<WinGetSource> sourceList = new List<WinGetSource>();
561
562 string[] jsonStrings = exportString.Split("}{");
563 StringBuilder jsonString = new StringBuilder();
564 for (int i = 0; i < jsonStrings.Length; i++)
565 {
566 if (!jsonStrings[i].StartsWith('{'))
567 {
568 jsonString.Append('{');
569 }
570
571 jsonString.Append(jsonStrings[i]);
572
573 if (!jsonStrings[i].EndsWith('}'))
574 {
575 jsonString.Append('}');
576 }
577
578 WinGetSource? source =
579 JsonHandler.StringToObject<WinGetSource>(jsonString.ToString());
580
581 if (source == null)
582 {
583 throw new WinGetActionFailedException("Exporting sources failed. Could not parse json.");
584 }
585
586 sourceList.Add(source);
587
588 jsonString.Clear();
589 }
590
591 return sourceList;
592 }
