66 private static List<WinGetPackage> CreatePackageListFromOutput(string[] output, int[] columnList, PackageAction action = PackageAction.Default, string? sourceName = null)
67 {
68 List<WinGetPackage> resultList = new List<WinGetPackage>();
69
70 if (columnList.Length < 3)
71 {
72 return resultList;
73 }
74
75 for (int i = 0; i < output.Length; i++)
76 {
77 // Stop parsing the output when the end of the list is reached.
78 if (string.IsNullOrWhiteSpace(output[i]) || output[i].Length < columnList[^1])
79 {
80 break;
81 }
82
83 // [var1..var2] : selects the index range from var1 to var2
84 // (eg. if var1 is 2 and var2 is 5, the selectet index range will be [2, 3, 4])
85 WinGetPackage package = new WinGetPackage()
86 {
87 PackageName = output[i][columnList[0]..columnList[1]].Trim(),
88 PackageId = output[i][columnList[1]..columnList[2]].Trim()
89 };
90
91 //Set version info depending on the column count.
92 if (columnList.Length > 3)
93 {
94 package.PackageVersion = output[i][columnList[2]..columnList[3]].Trim();
95 package.PackageAvailableVersion = output[i][columnList[2]..columnList[3]].Trim();
96 }
97 else
98 {
99 package.PackageVersion = output[i][columnList[2]..].Trim();
100 package.PackageAvailableVersion = output[i][columnList[2]..].Trim();
101 }
102
103 //Set information, depending on the action and the column count.
104 if ((action == PackageAction.UpgradeList || action == PackageAction.InstalledList || action == PackageAction.Search) && columnList.Length >= 5)
105 {
106 string availableVersion = output[i][columnList[3]..columnList[4]].Trim();
107 if (!string.IsNullOrWhiteSpace(availableVersion) && action != PackageAction.Search)
108 {
109 package.PackageAvailableVersion = availableVersion;
110 }
111
112 package.PackageSourceName = output[i][columnList[4]..].Trim();
113 }
114 else if((action == PackageAction.InstalledList || action == PackageAction.Search) && columnList.Length == 4)
115 {
116 package.PackageSourceName = output[i][columnList[3]..].Trim();
117 }
118 else if ((action == PackageAction.SearchBySource || action == PackageAction.InstalledListBySource) && !string.IsNullOrWhiteSpace(sourceName))
119 {
120 package.PackageSourceName = sourceName;
121 }
122
123 resultList.Add(package);
124 }
125
126 // Check for secondery list in output.
127 if (ArrayManager.GetEntryContains(output, "------") != -1)
128 {
129 List<WinGetPackage> seconderyList = ToPackageList(output, action);
130 resultList.AddRange(seconderyList);
131 }
132
133 return resultList;
134 }

201 private static List<WinGetSource> CreateSourceListFromOutput(string[] output, int[] columnList)
202 {
203 List<WinGetSource> resultList = new List<WinGetSource>();
204
205 if (columnList.Length < 2)
206 {
207 return resultList;
208 }
209
210 for (int i = 0; i < output.Length; i++)
211 {
212 // [var1..var2] : selects the index range from var1 to var2.
213 // (eg. if var1 is 2 and var2 is 5, the selectet index range will be [2, 3, 4])
214 // [var1..] : selects the index range from var1 to the end.
215 // (eg. if var1 is 2 and the last index is 5, the selectet index range will be[2, 3, 4, 5])
216 resultList.Add(
217 new WinGetSource()
218 {
219 SourceName = output[i][columnList[0]..columnList[1]].Trim(),
220 SourceUrl = output[i][columnList[1]..].Trim(),
221 SourceType = string.Empty
222 });
223 }
224
225 return resultList;
226 }

237 private static int[] GetColumnList(string line)
238 {
239 int[] columns = new int[0];
240
241 bool checkForColumn = true;
242 for (int i = 0; i < line.Length; i++)
243 {
244 if (line[i] != ((char)32) && checkForColumn)
245 {
246 columns = ArrayManager.Add(columns, i);
247 checkForColumn = false;
248 }
249 else if (line[i] == ((char)32))
250 {
251 checkForColumn = true;
252 }
253 }
254
255 return columns;
256 }
