65 private static List<WinGetPackage> CreatePackageListFromOutput(string[] output, int idStartIndex, int versionStartIndex, int extraInfoStartIndex)
66 {
67 List<WinGetPackage> resultList = new List<WinGetPackage>();
68
69 for (int i = 0; i < output.Length; i++)
70 {
71 // Stop parsing the output when the end of the list is reached.
72 if (string.IsNullOrWhiteSpace(output[i]) || output[i].Length < extraInfoStartIndex)
73 {
74 break;
75 }
76
77 // [var1..var2] : selects the index range from var1 to var2
78 // (eg. if var1 is 2 and var2 is 5, the selectet index range will be [2, 3, 4])
79 resultList.Add(
80 new WinGetPackage()
81 {
82 PackageName = output[i][0..idStartIndex].Trim(),
83 PackageId = output[i][idStartIndex..versionStartIndex].Trim(),
84 PackageVersion = output[i][versionStartIndex..extraInfoStartIndex].Trim()
85 });
86 }
87
88 // Check for secondery list in output.
89 if (ArrayManager.GetEntryContains(output, "------") != -1)
90 {
91 List<WinGetPackage> seconderyList = ToPackageList(output);
92 resultList.AddRange(seconderyList);
93 }
94
95 return resultList;
96 }

166 private static List<WinGetSource> CreateSourceListFromOutput(string[] output, int urlStartIndex)
167 {
168 List<WinGetSource> resultList = new List<WinGetSource>();
169
170 for (int i = 0; i < output.Length; i++)
171 {
172 // [var1..var2] : selects the index range from var1 to var2.
173 // (eg. if var1 is 2 and var2 is 5, the selectet index range will be [2, 3, 4])
174 // [var1..] : selects the index range from var1 to the end.
175 // (eg. if var1 is 2 and the last index is 5, the selectet index range will be[2, 3, 4, 5])
176 resultList.Add(
177 new WinGetSource()
178 {
179 SourceName = output[i][0..urlStartIndex].Trim(),
180 SourceUrl = output[i][urlStartIndex..].Trim(),
181 SourceType = string.Empty
182 });
183 }
184
185 return resultList;
186 }

201 private static int GetColumnStartIndex(string line, int column)
202 {
203 int currentColumn = 0;
204 bool checkForColumn = true;
205 for (int i = 0; i < line.Length; i++)
206 {
207 if (line[i] != ((char)32) && checkForColumn)
208 {
209 currentColumn++;
210 if (currentColumn == column)
211 {
212 return i;
213 }
214 checkForColumn = false;
215 }
216 else if (line[i] == ((char)32))
217 {
218 checkForColumn = true;
219 }
220 }
221 return -1;
222 }
