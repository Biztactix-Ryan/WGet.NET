967 private string AddArgumentByVersion(string argument)
968 {
969 // Checking version to determine if "--include-unknown" is necessary.
970 Version winGetVersion = WinGetVersionObject;
971 if (winGetVersion.Major >= 1 && winGetVersion.Minor >= 4)
972 {
973 // Winget version supports new argument, add "--include-unknown" to arguments
974 argument += $" {_includeUnknown}";
975 }
976 return argument;

1316 private string HashResultToHash(ProcessResult result)
1317 {
1318 string hash = "";
1319 if (result.Output.Length > 0 && result.Output[0].Contains(':'))
1320 {
1321 string[] splitOutput = result.Output[0].Split(':');
1322 if (splitOutput.Length >= 2)
1323 {
1324 hash = splitOutput[1].Trim();
1325 }
1326 }
1327
1328 return hash;
