1414 private string HashResultToHash(ProcessResult result)
1415 {
1416 string hash = "";
1417 if (result.Output.Length > 0 && result.Output[0].Contains(":"))
1418 {
1419 string[] splitOutput = result.Output[0].Split(':');
1420 if (splitOutput.Length >= 2)
1421 {
1422 hash = splitOutput[1].Trim();
1423 }
1424 }
1425
1426 return hash;
