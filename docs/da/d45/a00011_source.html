1417 private string HashResultToHash(ProcessResult result)
1418 {
1419 string hash = "";
1420 if (result.Output.Length > 0 && result.Output[0].Contains(":"))
1421 {
1422 string[] splitOutput = result.Output[0].Split(':');
1423 if (splitOutput.Length >= 2)
1424 {
1425 hash = splitOutput[1].Trim();
1426 }
1427 }
1428
1429 return hash;
1430 }
