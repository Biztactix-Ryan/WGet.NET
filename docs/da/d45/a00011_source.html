1354 private string HashResultToHash(ProcessResult result)
1355 {
1356 string hash = "";
1357 if (result.Output.Length > 0 && result.Output[0].Contains(":"))
1358 {
1359 string[] splitOutput = result.Output[0].Split(':');
1360 if (splitOutput.Length >= 2)
1361 {
1362 hash = splitOutput[1].Trim();
1363 }
1364 }
1365
1366 return hash;
1367 }
