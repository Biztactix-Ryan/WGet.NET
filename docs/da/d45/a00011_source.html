386 private string AddArgumentByVersion(string argument)
387 {
388 // Checking version to determine if "--include-unknown" is necessary
389 bool castSuccessful = int.TryParse(WinGetVersion.Split(".")[1], out int wingetMinorVersion);
390 if (castSuccessful && wingetMinorVersion >= 4)
391 {
392 // Winget version supports new argument, add "--include-unknown" to arguments
393 argument += " " + _includeUnknown;
394 }
395 return argument;
396 }

571 private string HashResultToHash(ProcessResult result)
572 {
573 string hash = "";
574 if (result.Output.Length > 0 && result.Output[0].Contains(':'))
575 {
576 string[] splitOutput = result.Output[0].Split(':');
577 if (splitOutput.Length >= 2)
578 {
579 hash = splitOutput[1].Trim();
580 }
581 }
582
583 return hash;
584 }
