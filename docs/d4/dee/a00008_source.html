240 private string CheckWinGetVersion()
241 {
242 try
243 {
244 ProcessResult result =
245 _processManager.ExecuteWingetProcess(_versionCmd);
246
247 for (int i = 0; i < result.Output.Length; i++)
248 {
249 if (result.Output[i].StartsWith("v"))
250 {
251 return result.Output[i].Trim();
252 }
253 }
254 }
255 catch
256 {
257 //No handling needed, winget can not be accessed
258 }
259
260 return string.Empty;
261 }

263 private Version GetVersionObject()
264 {
265 string versionString = CheckWinGetVersion();
266
267#if NETCOREAPP3_1_OR_GREATER
268 //Remove the first letter from the version string.
269 if (versionString.StartsWith('v'))
270 {
271 versionString = versionString[1..].Trim();
272
273 }
274
275 //Remove text from the end of the version string.
276 for (int i = 0; i < versionString.Length; i++)
277 {
278 if (versionString[i] == '-')
279 {
280 versionString = versionString[0..i];
281 break;
282 }
283 }
284#elif NETSTANDARD2_0
285 //Remove the first letter from the version string.
286 if (versionString.StartsWith("v"))
287 {
288 versionString = versionString.Substring(1).Trim();
289
290 }
291
292 //Remove text from the end of the version string.
293 for (int i = 0; i < versionString.Length; i++)
294 {
295 if (versionString[i] == '-')
296 {
297 versionString = versionString.Substring(0, i);
298 break;
299 }
300 }
301#endif
302
303 if (!Version.TryParse(versionString, out Version? versionObject) || versionObject == null)
304 {
305 versionObject = Version.Parse("0.0.0");
306 }
307
308 return versionObject;
309 }
