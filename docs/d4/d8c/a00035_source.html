46 public bool IsInstalled
47 {
48 get
49 {
50 // Check if the winget executable still exist to ensure a correct result,
51 // even if winget gets removed while the application is running.
52 if (File.Exists(_wingetExePath))
53 {
54 return true;

271 public List<WinGetAdminSetting> GetAdminSettings()
272 {
273 List<WinGetAdminSetting> adminSettings = new();
274
275 string settingsJson = ExportSettings();
276
277 SettingsModel settings = JsonHelper.StringToObject<SettingsModel>(settingsJson);
278 if (settings == null)
279 {
280 return adminSettings;
281 }
282
283 WinGetAdminSettingBuilder builder = new();
284 foreach (KeyValuePair<string, bool> entry in settings.AdminSettings)
285 {
286 builder.Clear();
287
288 builder.AddEntryName(entry.Key);
289 builder.AddStatus(entry.Value);
290
291 WinGetAdminSetting? adminSetting = builder.GetInstance();
292 if (adminSetting != null)
293 {
294 adminSettings.Add(adminSetting);
295 }
296 }
297
298 return adminSettings;
299 }

314 public async Task<List<WinGetAdminSetting>> GetAdminSettingsAsync(CancellationToken cancellationToken = default)
315 {
316 List<WinGetAdminSetting> adminSettings = new();
317
318 string settingsJson = await ExportSettingsAsync(cancellationToken);
319
320#if NETCOREAPP3_1_OR_GREATER
321 SettingsModel settings = await JsonHelper.StringToObjectAsync<SettingsModel>(settingsJson, cancellationToken);
322#elif NETSTANDARD2_0
323 SettingsModel settings = JsonHelper.StringToObject<SettingsModel>(settingsJson);
324#endif
325 if (settings == null)
326 {
327 return adminSettings;
328 }
329
330 WinGetAdminSettingBuilder builder = new();
331 foreach (KeyValuePair<string, bool> entry in settings.AdminSettings)
332 {
333 if (cancellationToken.IsCancellationRequested)
334 {
335 return adminSettings;
336 }
337
338 builder.Clear();
339
340 builder.AddEntryName(entry.Key);
341 builder.AddStatus(entry.Value);
342
343 WinGetAdminSetting? adminSetting = builder.GetInstance();
344 if (adminSetting != null)
345 {
346 adminSettings.Add(adminSetting);
347 }
348 }
349
350 return adminSettings;

620 public WinGetInfo GetInfo()
621 {
622 ProcessResult result = Execute(_infoCmd);
623
624 InfoActionVersionId actionVersionId = InfoActionVersionId.VersionRange1;
625 if (CheckWinGetVersion(new Version(1, 4, 3531), new Version(1, 5, 101)))
626 {
627 actionVersionId = InfoActionVersionId.VersionRange2;
628 }
629 else if (CheckWinGetVersion(new Version(1, 5, 441), new Version(1, 5, 441)))
630 {
631 actionVersionId = InfoActionVersionId.VersionRange3;
632 }
633 else if (CheckWinGetVersion(new Version(1, 5, 1081)))
634 {
635 actionVersionId = InfoActionVersionId.VersionRange4;
636 }
637
638 return ProcessOutputReader.ToWingetInfo(result.Output, actionVersionId);
639 }

654 public async Task<WinGetInfo> GetInfoAsync(CancellationToken cancellationToken = default)

659 InfoActionVersionId actionVersionId = InfoActionVersionId.VersionRange1;

662 actionVersionId = InfoActionVersionId.VersionRange2;
663 }

666 actionVersionId = InfoActionVersionId.VersionRange3;
667 }
668 else if (CheckWinGetVersion(new Version(1, 5, 1081)))
669 {
670 actionVersionId = InfoActionVersionId.VersionRange4;
671 }
672
673 // Return empty data if the task was canceled
674 if (cancellationToken.IsCancellationRequested)
675 {
676 return WinGetInfo.Empty;
677 }
678
679 return ProcessOutputReader.ToWingetInfo(result.Output, actionVersionId);

701 ((maxVersion == null) || (winGetVersion.Major <= maxVersion.Major && winGetVersion.Minor <= maxVersion.Minor &&

812 private string CheckWinGetVersion()
813 {
814 if (!IsInstalled)
815 {
816 return string.Empty;
817 }
818
819 ProcessResult result = Execute(_versionCmd);
820
821 for (int i = 0; i < result.Output.Length; i++)
822 {
823#if NETCOREAPP3_1_OR_GREATER
824 if (result.Output[i].StartsWith('v') && result.Output[i].Length >= 2)
825 {
826 // Return output without the 'v' at the start.
827 return result.Output[i].Trim()[1..];
828 }
829#elif NETSTANDARD2_0
830 if (result.Output[i].StartsWith("v") && result.Output[i].Length >= 2)
831 {
832 // Return output without the 'v' at the start.
833 return result.Output[i].Trim().Substring(1);
834 }
835#endif
836 }
837
838 return string.Empty;
839 }

847 private string CheckInstallation()
848 {
849 string? pathEnvVar = Environment.GetEnvironmentVariable("Path");
850 if (string.IsNullOrWhiteSpace(pathEnvVar))
851 {
852 return string.Empty;
853 }
854
855 string[] paths = pathEnvVar.Split(';');
856
857 string exePath;
858 for (int i = 0; i < paths.Length; i++)
859 {
860 exePath = Path.Combine(paths[i], "winget.exe");
861 if (File.Exists(exePath))
862 {
863 return exePath;
864 }
865 }
866
867 return string.Empty;
868 }

892 private bool QueryInstallation()
893 {
894 bool isInstalled;
895
896 _wingetExePath = CheckInstallation();
897
898 if (string.IsNullOrWhiteSpace(_wingetExePath))
899 {
900 isInstalled = false;
901 _processManager = new ProcessManager("winget");
902 }
903 else
904 {
905 isInstalled = true;
906 _processManager = new ProcessManager(_wingetExePath);
907 }
908
909 _wingetExeModificationData = GetLastModificationData();
910
911 _versionString = CheckWinGetVersion();
912 _version = VersionParser.Parse(_versionString);
913
914 return isInstalled;
915 }

924 private bool CheckAdministratorPrivileges()
925 {
926 if (Environment.OSVersion.Platform == PlatformID.Unix)
927 {
928 // Making sure windows related functions dont get called on none windows systems.
929 return true;
930 }
931
932 using WindowsIdentity? identity = WindowsIdentity.GetCurrent(false);
933
934 if (identity != null)
935 {
936 return new WindowsPrincipal(identity)
937 .IsInRole(WindowsBuiltInRole.Administrator);
938 }
939
940 return false;
941 }
